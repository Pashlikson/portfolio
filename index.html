// Apps Script (Google Sheets) ‚Äî SalesDrive YML ‚Üí Sheet
// ‚ö†Ô∏è –ß—É—Ç–ª–∏–≤—ñ –¥–∞–Ω—ñ –Ω–µ —Ö–∞—Ä–¥–∫–æ–¥–∏–º–æ: URL –±–µ—Ä–µ–º–æ –∑—ñ Script Properties.
function parseXml() {
  const doc = SpreadsheetApp.getActiveSpreadsheet();

  // --- sheet bootstrap ---
  let sheet;
  try {
    sheet = doc.getSheetByName('Settings');
    sheet.clear();
  } catch (e) {
    sheet = doc.insertSheet();
    sheet.setName('Settings');
  }

  // --- config (safe) ---
  const PROPS = PropertiesService.getScriptProperties();
  const YML_URL =
    PROPS.getProperty('SALES_DRIVE_YML_URL') ||
    'https://<your-domain>.salesdrive.me/export/yml/export.yml?publicKey=***';

  // --- fetch & parse ---
  const xmlText = UrlFetchApp.fetch(YML_URL, { muteHttpExceptions: true }).getContentText();
  const docXml  = XmlService.parse(xmlText);
  const root    = docXml.getRootElement();        // <yml_catalog>
  const shop    = root.getChildren()[0];          // <shop>

  const offers     = shop.getChildren('offers')[0].getChildren('offer');
  const categories = shop.getChildren('categories')[0].getChildren('category');

  // map –∫–∞—Ç–µ–≥–æ—Ä—ñ–π (–Ω–∞ –º–∞–π–±—É—Ç–Ω—î)
  const categoryById = {};
  categories.forEach(c => {
    categoryById[c.getAttribute('id').getValue()] = c.getText();
  });

  // --- header ---
  const rows = [];
  rows.push(['–∫–æ–¥ —Ç–æ–≤–∞—Ä—É', '—Ü—ñ–Ω–∞', '—Å–æ–±—ñ–≤–∞—Ä—Ç—ñ—Å—Ç—å', '–∫—ñ–ª—å–∫—ñ—Å—Ç—å']);

  // --- offers ‚Üí rows ---
  offers.forEach(o => {
    const vendorCode  = o.getChildText('vendorCode') || '';
    const price       = o.getChildText('price') || '';
    const vendorPrice = o.getChildText('vendorprice') || '';
    const qty         = o.getChildText('quantity_in_stock') || ''; // ‚úÖ
    rows.push([vendorCode, price, vendorPrice, qty]);
  });

  // --- write to sheet ---
  if (rows.length) {
    sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
  }

  // --- timestamp ---
  const ts = Utilities.formatDate(new Date(), 'GMT+3', 'dd.MM.yyyy HH:mm');
  sheet.getRange('F1').setValue('–û–Ω–æ–≤–ª–µ–Ω–æ: ' + ts);
}

import os, time, requests
from datetime import datetime, timedelta, timezone
from collections import Counter

SD_BASE = os.getenv("SALES_DRIVE_API_BASE", "https://.salesdrive.me/api")
SD_KEY  = os.getenv("SALES_DRIVE_API_KEY",  "sd_***")
TG_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN",  "6xxx:***")
TG_CHAT  = os.getenv("TELEGRAM_CHAT_ID",    "123456789")

HDRS = {"Authorization": f"Bearer {SD_KEY}", "Content-Type": "application/json"}
KYIV = timezone(timedelta(hours=3))

def date_range_yesterday():
    end = datetime.now(KYIV).replace(hour=0, minute=0, second=0, microsecond=0)
    start = end - timedelta(days=1)
    # ISO –≤ UTC –¥–ª—è API
    return (start.astimezone(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
            end.astimezone(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
            start.strftime("%d.%m.%Y"))

def sd_get_orders(date_from, date_to):
    """–õ–∞–∫–æ–Ω—ñ—á–Ω–∏–π –∑–∞–ø–∏—Ç –∑ –º‚Äô—è–∫–∏–º —Ä–µ—Ç—Ä–∞—î–º —Ç–∞ –ø—Ä–æ—Å—Ç–æ—é –ø–∞–≥—ñ–Ω–∞—Ü—ñ—î—é."""
    page, out = 1, []
    while True:
        url = f"{SD_BASE.rstrip('/')}/orders"
        params = {"dateFrom": date_from, "dateTo": date_to, "status": "completed", "page": page, "limit": 500}
        backoff = 1.0
        for _ in range(3):
            try:
                r = requests.get(url, headers=HDRS, params=params, timeout=20)
                if r.status_code in (429, 502, 503, 504): time.sleep(backoff); backoff *= 2; continue
                r.raise_for_status()
                data = r.json()
                chunk = data.get("orders") or data.get("items") or []
                out.extend(chunk)
                if not data.get("nextPage") and page >= data.get("pages", page): return out
                page += 1
                break
            except requests.RequestException:
                time.sleep(backoff); backoff *= 2
    return out

def aggregate(orders):
    revenue = 0.0
    items = Counter()
    for o in orders:
        revenue += float(o.get("total") or o.get("sum") or o.get("grandTotal") or 0)
        for it in (o.get("items") or o.get("products") or []):
            name = it.get("name") or it.get("title") or (it.get("sku") or it.get("article") or "item")
            qty  = int(it.get("qty") or it.get("quantity") or 1)
            items[name] += qty
    orders_count = len(orders)
    avg = revenue / orders_count if orders_count else 0.0
    return revenue, orders_count, avg, items.most_common(5)

    s = f"{v:,.2f}".replace(",", " ").replace(".", ",")
    return f"{s} –≥—Ä–Ω"

        f"‚îÉ –ó–∞–º–æ–≤–ª–µ–Ω—å: *{orders_cnt}*\n"
        f"‚îÉ –°–µ—Ä–µ–¥–Ω—ñ–π —á–µ–∫: *{fmt_uah(avg)}*\n"

def tg_send_md(msg):
    url = f"https://api.telegram.org/bot{TG_TOKEN}/sendMessage"
    requests.post(url, json={"chat_id": TG_CHAT, "text": msg, "parse_mode": "Markdown"}).raise_for_status()

def main():
